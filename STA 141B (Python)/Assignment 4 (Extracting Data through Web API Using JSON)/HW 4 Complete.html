def ndb_search_url(search_term, data_source, max_per_page, key, offset = 0, verbose = False):
    """This function takes in a search term(str), the max amount of values to be returned (int) and the 
    API key for data.gov(str). Then outputs the matching results in a json file."""
    
    #Determine the search specficiations for the API
    url = "https://api.nal.usda.gov/ndb/search"
    response = requests.get(url, params = {
            "q" : search_term,
            "format" : "json",
            "sort" : "r",
            "ds": data_source,
            "max" : max_per_page,
            "offset" : offset,
            "api_key" : key
        })
    response.raise_for_status() # check for errors
    if verbose:
        print response.url
    return response.json()[u'list'][u'item'] # parse JSON
